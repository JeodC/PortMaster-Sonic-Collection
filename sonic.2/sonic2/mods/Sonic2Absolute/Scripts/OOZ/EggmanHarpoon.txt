event ObjectMain
	switch object.state
	case 0
		if object.propertyValue == 0
			object.value1 = object.xpos
			object.value1 += 0x800000
			object.value2 = object.ypos
			object.state = 2
		else
			object.value1 = object.xpos
			object.value1 -= 0x800000
			object.value2 = object.ypos
			object.state = 1
		end if
		temp1 = object.angle
		break
	case 1
		object.angle += 2
		object.angle &= 511
		if object.angle == 0
			object.type = TypeName[Blank Object]
		end if
		temp1 = object.angle
		temp1 += 12
		break
	case 2
		object.angle -= 2
		object.angle &= 511
		if object.angle == 0
			object.type = TypeName[Blank Object]
		end if
		temp1 = object.angle
		temp1 -= 12
		break
	end switch
	Sin(object.xpos, object.angle)
	object.xpos *= 0x3400
	object.xpos += object.value1
	Cos(object.ypos, object.angle)
	object.ypos *= 0x3400
	object.ypos += object.value2
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		BoxCollisionTest(C_TOUCH, object.entityPos, -8, -8, 8, 8, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
		if checkResult == 1
			CallFunction(PlayerObject_Hit)
		end if
	next
	Sin(object.xpos, temp1)
	object.xpos *= 0x3400
	object.xpos += object.value1
	Cos(object.ypos, temp1)
	object.ypos *= 0x3400
	object.ypos += object.value2
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		BoxCollisionTest(C_TOUCH, object.entityPos, -8, -8, 8, 8, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
		if checkResult == 1
			CallFunction(PlayerObject_Hit)
		end if
	next
end event


event ObjectDraw
	temp1 = object.angle
	if object.state == 2
		temp1 += 108
		temp1 &= 511
		temp0 = 0
		while temp0 < 9
			Sin(object.xpos, temp1)
			object.xpos *= 0x3400
			object.xpos += object.value1
			Cos(object.ypos, temp1)
			object.ypos *= 0x3400
			object.ypos += object.value2
			DrawSprite(0)
			temp0++
			temp1 -= 12
			temp1 &= 511
		loop
		object.direction = FLIP_X
		object.rotation = 384
		object.rotation -= object.angle
		object.rotation &= 511
	else
		temp1 -= 108
		temp1 &= 511
		temp0 = 0
		while temp0 < 9
			Sin(object.xpos, temp1)
			object.xpos *= 0x3400
			object.xpos += object.value1
			Cos(object.ypos, temp1)
			object.ypos *= 0x3400
			object.ypos += object.value2
			DrawSprite(0)
			temp0++
			temp1 += 12
			temp1 &= 511
		loop
		object.direction = FLIP_NONE
		object.rotation = 128
		object.rotation -= object.angle
		object.rotation &= 511
	end if
	DrawSpriteFX(1, FX_ROTATE, object.xpos, object.ypos)
end event
