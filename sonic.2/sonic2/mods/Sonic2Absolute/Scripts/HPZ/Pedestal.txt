event ObjectMain
	if object.state == 0
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			if currentPlayer == 0
				BoxCollisionTest(C_TOUCH, object.entityPos, -7, -7, 7, 7, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
				if checkResult == 1
					object.state = 1
					SetBit(specialStage.emeralds, 8, 1) //Eighth bit as a flag. We still have all the emeralds, but we've put down specific ones
					SetBit(specialStage.emeralds, object.propertyValue, 0) //StageSetup will return any emeralds on level load
					if PlayerObject_SuperState == 1
						PlayerObject_SuperState = 2
					end if
					PlaySfx(SfxName[Catch], 0)
				end if
			end if
		next
	end if
end event


event ObjectDraw
	DrawSprite(0)
	if object.state == 1
		DrawSprite(object.propertyValue)
	end if
end event

	SpriteFrame(-6, -6, 12, 12, 41, 73) // 7 white
	
	SetTableValue(TypeName[Pedestal], DebugMode_ObjCount, DebugMode_TypesTable)

	SetTableValue(Pedestal_DebugSpawn, DebugMode_ObjCount, DebugMode_SpawnTable)
	DebugMode_ObjCount++
	
end event
