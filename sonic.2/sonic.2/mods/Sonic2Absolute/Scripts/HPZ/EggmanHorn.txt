event ObjectMain
	switch object.state
	case 1
		object.frame = object.value0
		object.frame &= 3
		if object.frame == 0
			screen.shakeY = 2
		end if
		object.frame >>= 1
		if object.value0 > 0
			object.value0--
		else
			object.state = 0
		end if
		break
	case 2
		if object.value0 > 0
			object.frame = 1
			object.value0--
		else
			object.frame = 0
			object.state = 0
		end if
		break
	case 3
		object.frame = 0
		object.xpos += object.xvel
		object.ypos += object.yvel
		object.yvel += 0x3800
		if object.outOfBounds == 1
			object.type = TypeName[Blank Object]
		end if
		break
	end switch
end event


event ObjectDraw
	DrawSpriteFX(2, FX_FLIP, object.xpos, object.ypos)
	DrawSpriteFX(object.frame, FX_FLIP, object.xpos, object.ypos)
end event
