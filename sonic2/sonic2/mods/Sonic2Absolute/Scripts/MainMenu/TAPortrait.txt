event ObjectMain
	foreach (TypeName[Time Attack Screen], arrayPos0, ALL_ENTITIES)
		if object[arrayPos0].value10 == object.propertyValue
			switch object[arrayPos0].state
				case 0
				case 100
					if screen.xsize == 320
						if object.propertyValue == 2
							object.value0 = 75
						else	
							object.value0 = 80
						end if	
					else
						object.value0 = 100
					end if
					break	
				case 1
				case 2
					object.value0 = -360
				break
				default
					if screen.xsize == 320
						if object.propertyValue == 2
							object.value0 = 75
						else	
							object.value0 = 80
						end if
					else
						object.value0 = 100
					end if
					break
			end switch
		else
			object.value0 = -300
		end if
		
		switch object[arrayPos0].state
			case 0
			case 100
				object.value4 = 140
				break
			default
				object.value4 = 140
				break
		end switch
	next
	
	temp0 = object.value1
	temp0 -= object.value0
	Abs(temp0)
	if temp0 < charspeed
		object.value1 = object.value0
	end if
	if object.value1 > object.value0
		object.value1 -= charspeed
	else
		if object.value1 < object.value0
			object.value1 += charspeed
		end if
	end if
	
	temp0 = object.value3
	temp0 -= object.value4
	Abs(temp0)
	if temp0 < charspeed
		object.value3 = object.value4
	end if
	
	if object.value3 > object.value4
		object.value3 -= charspeed
	else
		if object.value3 < object.value4
			object.value3 += charspeed
		end if
	end if	
end event

event ObjectDraw
	temp0 = object.value3
	temp0 -= 55
	temp1 = object.value1
	temp1 += 20
	if screen.xsize == 320
		temp1 -= 18
	end if	
	DrawSpriteScreenXY(object.propertyValue, temp1, temp0)
	temp2 = object.propertyValue
	temp2 += 4
	temp0 = object.value3
	temp0 += 160
	temp1 = object.value1
	temp1 += 20
	if screen.xsize == 320
		temp1 -= 18
	end if
	DrawSpriteScreenXY(temp2, temp1, temp0)
end event
