sub ObjectMain
	switch Object.State
	case CRUSHBLOCKS_DELAY_1
		if Object.Timer < 60
			Object.Timer++
		else
			Object.Timer   = 0
			Object.XOffSet = 0x8000
			Object.State++
		end if
		break

	case CRUSHBLOCKS_OPEN
		if Object.XVelocity < 0x600000
			Object.XVelocity += 0x8000
		else
			Object.XOffSet = 0
			Object.State++
		end if
		break

	case CRUSHBLOCKS_DELAY_2
		if Object.Timer < 120
			Object.Timer++
		else
			Object.Timer   =  0
			Object.XOffSet = -0x20000
			Object.State++
		end if
		break

	case CRUSHBLOCKS_CRUSH
		if Object.XVelocity > 0
			Object.XVelocity -= 0x20000
		else
			Object.XOffSet = 0
			Object.State   = CRUSHBLOCKS_DELAY_1
		end if
		break

	end switch
end sub
