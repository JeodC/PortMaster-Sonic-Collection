sub ObjectPlayerInteraction
	if Object.PropertyValue == 0
		if Player.State != Player_State_Static
			PlayerObjectCollision(C_TOUCH, -16, -16, 16, 16)
			if CheckResult == true
				Player.State			 = Player_State_Static
				Player.ObjectInteraction = false
				Player.Visible			 = false

				if Player.BlinkTimer > 0
					Player.InvincibleTimer	= 0
					Player.BlinkTimer		= 0
				end if

				Object.State	 = TUNNELPATH_1_ACTIVE
				Object.MoveSteps = -1
				Object.Priority	 = PRIORITY_ACTIVE

				PlaySfx(SFX_G_RELEASE, false)
			end if
		end if
	end if
end sub

sub RSDKDraw
	Object.InkEffect = 1
	DrawSpriteFX(0,FX_INK,Object.XPos, Object.YPos)
	if Editor.ShowGizmos == true
		Editor.DrawingOverlay = true

		CallFunction(EditorHelpers_FindTimePeriod)
		switch CheckResult
		case 0	// Present
		case 3	// Bad Future
			TempValue6 = 255
			TempValue7 = 0
			break
		case 1	// Past
			TempValue6 = 0
			TempValue7 = 128
			break
		case 2	// Good Future
			TempValue6 = 128
			TempValue7 = 0
			break
		end switch

		TempValue0 = 0
		while TempValue0 <= 8
			TempValue1  = Object.XPos
			TempValue2  = Object.YPos
			TempValue3  = Object.XPos
			TempValue4  = Object.YPos
			TempValue5  = 0
			switch TempValue0
			case 0
				TempValue1 += 0x2000
				TempValue3 += 0x2000
				TempValue2 += 0x1500
				TempValue4 += 0x1501500
				TempValue5  = 1
				break
			case 1
				TempValue1 += 0x2000
				TempValue3 -= 0x67E000
				TempValue2 += 0x1501500
				TempValue4 += 0x1501500
				break
			case 2
				TempValue1 -= 0x67E000
				TempValue3 -= 0x67E000
				TempValue2 += 0x1501500
				TempValue4 += 0x2001500
				TempValue5  = 1
				break
			case 3
				TempValue1 -= 0x67E000
				TempValue3 -= 0xFE000
				TempValue2 += 0x2001500
				TempValue4 += 0x2001500
				break
			case 4
				TempValue1 -= 0xFE000
				TempValue3 -= 0x37E000
				TempValue2 += 0x2081500
				TempValue4 += 0x2081500
				break
			case 5
				TempValue1 -= 0x37E000
				TempValue3 -= 0x37E000
				TempValue2 += 0x2081500
				TempValue4 += 0x3281500
				TempValue5  = 1
				break
			case 6
				TempValue1 -= 0x37E000
				TempValue3 += 0x2182000
				TempValue2 += 0x3281500
				TempValue4 += 0x3281500
				break
			case 7
				TempValue1 += 0x2182000
				TempValue3 += 0x2182000
				TempValue2 += 0x3281500
				TempValue4 += 0x3701500
				TempValue5  = 1
				break
			case 8
				TempValue1 += 0x2182000
				TempValue3 += 0x3C82000
				TempValue2 += 0x3701500
				TempValue4 += 0x3701500
				break
			endswitch
			DrawArrow(TempValue1, TempValue2, TempValue3, TempValue4, 255, TempValue6, TempValue7)
			// make the arrow thicker
			if TempValue5 == 1
				TempValue1 += 0x10000
				TempValue3 += 0x10000
				DrawArrow(TempValue1, TempValue2, TempValue3, TempValue4, 255, TempValue6, TempValue7)
			else
				TempValue2 -= 0x10000
				TempValue4 -= 0x10000
				DrawArrow(TempValue1, TempValue2, TempValue3, TempValue4, 255, TempValue6, TempValue7)
			end if
			TempValue0++
		loop
		Editor.DrawingOverlay = false
	end if
end sub
